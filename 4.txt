// BÖLÜM 4: ZIGZAG MTF MARKET YAPISI MOTORU + DİNAMİK FİB
// Gecikmesiz swing tespiti — highestbars/lowestbars kullanır
// request.security gerektirmez — MTF desteği time() ile gömülü
// ═══════════════════════════════════════════════════════════
ms_zigzag(string tf, int prd, bool useCloseConf) =>
    // ─── MTF Pencere Hesabı ─────────────────────────────────
    bool newbar = ta.change(time(tf)) != 0
    int bi = ta.valuewhen(newbar, bar_index, prd - 1)
    int len = math.max(bar_index - nz(bi, bar_index) + 1, 1)

    // ─── Anlık Swing Tespiti ────────────────────────────────
    float _ph = ta.highestbars(high, len) == 0 ? high : na
    float _pl = ta.lowestbars(low, len) == 0 ? low : na

    // ─── ZigZag Yön ve Swing Takibi ────────────────────────
    var int dir = 0
    var float zz_curr = na,   var int zz_curr_t = na,   var float zz_curr_c = na    // Canlı uç
    var float zz_p1 = na,     var int zz_p1_t = na,     var float zz_p1_c = na      // Son onaylı nokta
    var float zz_p2 = na,     var int zz_p2_t = na,     var float zz_p2_c = na      // Bir önceki onaylı
    var float zz_p3 = na,     var int zz_p3_t = na,     var float zz_p3_c = na      // İki önceki onaylı

    int prev_dir = dir
    dir := not na(_ph) and na(_pl) ? 1 : not na(_pl) and na(_ph) ? -1 : dir
    bool dirchanged = dir != prev_dir

    if dirchanged
        // Yön değişti → eski canlı ucu onayla, yeni canlı uç başlat
        zz_p3 := zz_p2
        zz_p3_t := zz_p2_t
        zz_p3_c := zz_p2_c
        zz_p2 := zz_p1
        zz_p2_t := zz_p1_t
        zz_p2_c := zz_p1_c
        zz_p1 := zz_curr
        zz_p1_t := zz_curr_t
        zz_p1_c := zz_curr_c
        zz_curr := dir == 1 ? high : low
        zz_curr_t := time
        zz_curr_c := dir == 1 ? math.max(open, close) : math.min(open, close)
    else
        // Aynı yön → canlı ucu güncelle
        if dir == 1
            if na(zz_curr) or high >= zz_curr
                zz_curr := high
                zz_curr_t := time
                zz_curr_c := math.max(open, close)
        else if dir == -1
            if na(zz_curr) or low <= zz_curr
                zz_curr := low
                zz_curr_t := time
                zz_curr_c := math.min(open, close)

    // ─── Onaylı Swing High/Low ──────────────────────────────
    var float conf_high = na,  var int conf_high_t = na, var float conf_high_c = na
    var float conf_low = na,   var int conf_low_t = na,  var float conf_low_c = na
    var float prev_conf_high = na, var int prev_conf_high_t = na, var float prev_conf_high_c = na
    var float prev_conf_low = na,  var int prev_conf_low_t = na,  var float prev_conf_low_c = na
    var bool high_active = false
    var bool low_active = false

    float _nsh = na, float _nsl = na
    int _nsh_t = na, int _nsl_t = na
    int _swing_type = 0

    // QML Tracker Variables
    var float qml_bull_L1 = na, var int qml_bull_L1_t = na, var float qml_bull_L1_c = na
    var float qml_bull_H1 = na, var int qml_bull_H1_t = na
    var float qml_bull_L2 = na, var int qml_bull_L2_t = na
    var bool  qml_bull_armed = false

    var float qml_bear_H1 = na, var int qml_bear_H1_t = na, var float qml_bear_H1_c = na
    var float qml_bear_L1 = na, var int qml_bear_L1_t = na
    var float qml_bear_H2 = na, var int qml_bear_H2_t = na
    var bool  qml_bear_armed = false

    if dirchanged
        if dir == 1  // Aşağı gidiyordu, yukarı döndü → LOW onaylandı
            prev_conf_low := conf_low
            prev_conf_low_t := conf_low_t
            prev_conf_low_c := conf_low_c
            conf_low := zz_p1
            conf_low_t := zz_p1_t
            conf_low_c := zz_p1_c
            low_active := true
            _swing_type := na(prev_conf_low) ? -2 : conf_low <= prev_conf_low ? -2 : -1
            _nsl := conf_low
            _nsl_t := conf_low_t
            
            // Bullish QML Check (L1 -> H1 -> L2)
            if not na(prev_conf_low) and conf_low < prev_conf_low
                qml_bull_armed := true
                qml_bull_L1 := prev_conf_low
                qml_bull_L1_t := prev_conf_low_t
                qml_bull_L1_c := prev_conf_low_c
                qml_bull_H1 := conf_high
                qml_bull_H1_t := conf_high_t
                qml_bull_L2 := conf_low
                qml_bull_L2_t := conf_low_t
            else
                qml_bull_armed := false

        else  // Yukarı gidiyordu, aşağı döndü → HIGH onaylandı
            prev_conf_high := conf_high
            prev_conf_high_t := conf_high_t
            prev_conf_high_c := conf_high_c
            conf_high := zz_p1
            conf_high_t := zz_p1_t
            conf_high_c := zz_p1_c
            high_active := true
            _swing_type := na(prev_conf_high) ? 2 : conf_high >= prev_conf_high ? 2 : 1
            _nsh := conf_high
            _nsh_t := conf_high_t

            // Bearish QML Check (H1 -> L1 -> H2)
            if not na(prev_conf_high) and conf_high > prev_conf_high
                qml_bear_armed := true
                qml_bear_H1 := prev_conf_high
                qml_bear_H1_t := prev_conf_high_t
                qml_bear_H1_c := prev_conf_high_c
                qml_bear_L1 := conf_low
                qml_bear_L1_t := conf_low_t
                qml_bear_H2 := conf_high
                qml_bear_H2_t := conf_high_t
            else
                qml_bear_armed := false

    // ─── BOS / CHoCH Detection & MS QML Engine ──────────────
    var int prevBreakDir = 0
    var int _bias = 0
    int _brk = 0
    float _brk_lvl = na
    int _brk_t = na
    float _highSrc = useCloseConf ? close : high
    float _lowSrc = useCloseConf ? close : low

    // ─── Makro Trend (Dealing Range) Takibi ─────────────────
    var float trend_origin_lvl = na, var int trend_origin_t = na, var float trend_origin_c = na
    var float trend_extreme_lvl = na, var int trend_extreme_t = na, var float trend_extreme_c = na
    var float prev_macro_top = na, var int prev_macro_top_t = na, var float prev_macro_top_c = na
    var float prev_macro_bot = na, var int prev_macro_bot_t = na, var float prev_macro_bot_c = na

    int prev_bias = _bias
    
    float qml_bull = na, int qml_bull_t = na, float qml_bull_c = na, float qml_bull_n = na, int qml_bull_c_t = na, float qml_bull_h = na, int qml_bull_h_t = na
    float qml_bear = na, int qml_bear_t = na, float qml_bear_c = na, float qml_bear_n = na, int qml_bear_c_t = na, float qml_bear_h = na, int qml_bear_h_t = na

    if not na(conf_high) and high_active and _highSrc > conf_high
        _brk := prevBreakDir == -1 ? 2 : 1
        _brk_lvl := conf_high
        _brk_t := conf_high_t
        prevBreakDir := 1
        _bias := 1
        high_active := false
        
        // Bullish QML Trigger
        if _brk == 2 and qml_bull_armed and conf_high == qml_bull_H1
            qml_bull := qml_bull_L1
            qml_bull_t := qml_bull_L1_t
            qml_bull_c := qml_bull_L1_c
            qml_bull_n := qml_bull_H1
            qml_bull_c_t := qml_bull_H1_t
            qml_bull_h := qml_bull_L2
            qml_bull_h_t := qml_bull_L2_t
            qml_bull_armed := false

    if not na(conf_low) and low_active and _lowSrc < conf_low
        _brk := prevBreakDir == 1 ? -2 : -1
        _brk_lvl := conf_low
        _brk_t := conf_low_t
        prevBreakDir := -1
        _bias := -1
        low_active := false
        
        // Bearish QML Trigger
        if _brk == -2 and qml_bear_armed and conf_low == qml_bear_L1
            qml_bear := qml_bear_H1
            qml_bear_t := qml_bear_H1_t
            qml_bear_c := qml_bear_H1_c
            qml_bear_n := qml_bear_L1
            qml_bear_c_t := qml_bear_L1_t
            qml_bear_h := qml_bear_H2
            qml_bear_h_t := qml_bear_H2_t
            qml_bear_armed := false
        
    if _bias != prev_bias
        if prev_bias == 0
            // İlk trend tespiti — makro değerleri başlat
            if _bias == 1
                trend_origin_lvl := conf_low
                trend_origin_t := conf_low_t
                trend_origin_c := conf_low_c
                trend_extreme_lvl := conf_high
                trend_extreme_t := conf_high_t
                trend_extreme_c := conf_high_c
            else if _bias == -1
                trend_origin_lvl := conf_high
                trend_origin_t := conf_high_t
                trend_origin_c := conf_high_c
                trend_extreme_lvl := conf_low
                trend_extreme_t := conf_low_t
                trend_extreme_c := conf_low_c
        else
            // Gerçek bir CHoCH oldu — eski trendi kaydet
            if _bias == 1
                prev_macro_top := trend_origin_lvl
                prev_macro_top_t := trend_origin_t
                prev_macro_top_c := trend_origin_c
                prev_macro_bot := trend_extreme_lvl
                prev_macro_bot_t := trend_extreme_t
                prev_macro_bot_c := trend_extreme_c
                trend_origin_lvl := trend_extreme_lvl
                trend_origin_t := trend_extreme_t
                trend_origin_c := trend_extreme_c
                trend_extreme_lvl := conf_high
                trend_extreme_t := conf_high_t
                trend_extreme_c := conf_high_c
            else if _bias == -1
                prev_macro_top := trend_extreme_lvl
                prev_macro_top_t := trend_extreme_t
                prev_macro_top_c := trend_extreme_c
                prev_macro_bot := trend_origin_lvl
                prev_macro_bot_t := trend_origin_t
                prev_macro_bot_c := trend_origin_c
                trend_origin_lvl := trend_extreme_lvl
                trend_origin_t := trend_extreme_t
                trend_origin_c := trend_extreme_c
                trend_extreme_lvl := conf_low
                trend_extreme_t := conf_low_t
                trend_extreme_c := conf_low_c
    else
        if _bias == 1
            if not na(conf_high) and conf_high > nz(trend_extreme_lvl, 0)
                trend_extreme_lvl := conf_high
                trend_extreme_t := conf_high_t
                trend_extreme_c := conf_high_c
        else if _bias == -1
            if not na(conf_low) and conf_low < nz(trend_extreme_lvl, 999999)
                trend_extreme_lvl := conf_low
                trend_extreme_t := conf_low_t
                trend_extreme_c := conf_low_c

    // ─── Fibonacci: Canlı Makro (Active Dealing Range) ──────────────────
    float fib_live_top = na, float fib_live_bot = na
    int fib_live_top_t = na, int fib_live_bot_t = na

    if _bias != 0 and not na(trend_origin_lvl)
        float live_extreme = trend_extreme_lvl
        int live_extreme_t = trend_extreme_t
        if _bias == 1
            if dir == 1 and not na(zz_curr) and (na(live_extreme) or zz_curr > live_extreme)
                live_extreme := zz_curr
                live_extreme_t := zz_curr_t
            if high > live_extreme
                live_extreme := high
                live_extreme_t := time
        else if _bias == -1
            if dir == -1 and not na(zz_curr) and (na(live_extreme) or zz_curr < live_extreme)
                live_extreme := zz_curr
                live_extreme_t := zz_curr_t
            if low < live_extreme
                live_extreme := low
                live_extreme_t := time
            
        if not na(live_extreme)
            fib_live_top := math.max(trend_origin_lvl, live_extreme)
            fib_live_bot := math.min(trend_origin_lvl, live_extreme)
            fib_live_top_t := trend_origin_lvl >= live_extreme ? trend_origin_t : live_extreme_t
            fib_live_bot_t := trend_origin_lvl < live_extreme ? trend_origin_t : live_extreme_t

    // ─── Fibonacci: Kırılan Makro (Broken Dealing Range) ────────────────
    float fib_broken_top = na, float fib_broken_bot = na
    int fib_broken_top_t = na, int fib_broken_bot_t = na

    if not na(prev_macro_top) and not na(prev_macro_bot)
        fib_broken_top := prev_macro_top
        fib_broken_bot := prev_macro_bot
        fib_broken_top_t := prev_macro_top_t
        fib_broken_bot_t := prev_macro_bot_t

    [_bias, _brk, _brk_lvl, _brk_t, conf_high, conf_low, conf_high_t, conf_low_t, _nsh, _nsl, _nsh_t, _nsl_t, _swing_type, fib_live_top, fib_live_bot, fib_live_top_t, fib_live_bot_t, fib_broken_top, fib_broken_bot, fib_broken_top_t, fib_broken_bot_t, qml_bull, qml_bull_t, qml_bull_c, qml_bull_n, qml_bull_c_t, qml_bull_h, qml_bull_h_t, qml_bear, qml_bear_t, qml_bear_c, qml_bear_n, qml_bear_c_t, qml_bear_h, qml_bear_h_t]

f_add_qml(float lvl, int t, float c, float n, int c_t, float head, int h_t, bool is_bull, bool show_visual, string tf_label) =>
    bool skip = false
    if qmlShoulderProtect > 0 and array.size(qml_zones) > 0
        for i = array.size(qml_zones) - 1 to 0
            QMLZone q = array.get(qml_zones, i)
            if q.is_valid and q.is_bull == is_bull
                if (bar_index - q.b_idx) <= qmlShoulderProtect
                    bool is_better = is_bull ? (head < q.head) : (head > q.head)
                    if is_better
                        q.is_valid := false
                        if not na(q.b)
                            box.delete(q.b)
                            label.delete(q.l)
                        if array.size(q.lines) > 0
                            for j = 0 to array.size(q.lines) - 1
                                line.delete(array.get(q.lines, j))
                        if array.size(q.labels) > 0
                            for j = 0 to array.size(q.labels) - 1
                                label.delete(array.get(q.labels, j))
                    else
                        skip := true
                    break
                    
    if not skip
        box bx = na
        label lb = na
        array<line> qml_lines = array.new<line>()
        array<label> qml_labels = array.new<label>()
        if show_visual
            color col = is_bull ? msBullColor : msBearColor
            string style_lbl = is_bull ? label.style_label_up : label.style_label_down
            float top_y = is_bull ? c : lvl
            float bot_y = is_bull ? lvl : c
            bx := box.new(left=t, top=top_y, right=time, bottom=bot_y, xloc=xloc.bar_time, border_color=color.new(col, 50), border_width=1, border_style=line.style_dashed, bgcolor=color.new(col, 85))
            lb := label.new(x=time, y=lvl, text="⚡ QML " + tf_label, xloc=xloc.bar_time, color=color.new(color.white, 100), textcolor=col, style=style_lbl, size=size.small)
            
            // Yapı Çizimi (LS -> Neck -> Head -> CHoCH)
            if showQmlStructure
                array.push(qml_lines, line.new(t, lvl, c_t, n, xloc=xloc.bar_time, color=col, width=1, style=line.style_dotted))
                array.push(qml_lines, line.new(c_t, n, h_t, head, xloc=xloc.bar_time, color=col, width=1, style=line.style_dotted))
                // Kırılan boyun çizgisini yatay olarak göster
                array.push(qml_lines, line.new(c_t, n, time, n, xloc=xloc.bar_time, color=col, width=2, style=line.style_solid))
                array.push(qml_labels, label.new(t, lvl, "LS", xloc=xloc.bar_time, color=color.new(color.black, 100), textcolor=col, size=size.tiny, style=style_lbl))
                array.push(qml_labels, label.new(c_t, n, "N", xloc=xloc.bar_time, color=color.new(color.black, 100), textcolor=col, size=size.tiny, style=is_bull ? label.style_label_down : label.style_label_up))
                array.push(qml_labels, label.new(h_t, head, "H", xloc=xloc.bar_time, color=color.new(color.black, 100), textcolor=col, size=size.tiny, style=style_lbl))

        array.push(qml_zones, QMLZone.new(lvl, t, bar_index, c, head, is_bull, true, bx, lb, tf_label, qml_lines, qml_labels))

f_get_closest_valid_qml(float entry_price, bool search_bull) =>
    float closest_lvl = na
    float min_dist = 999999.0
    if array.size(qml_zones) > 0
        for i = 0 to array.size(qml_zones) - 1
            QMLZone q = array.get(qml_zones, i)
            if q.is_valid and q.is_bull == search_bull
                if search_bull and q.c < entry_price
                    float dist = entry_price - q.c
                    if dist < min_dist
                        min_dist := dist
                        closest_lvl := q.c
                else if not search_bull and q.c > entry_price
                    float dist = q.c - entry_price
                    if dist < min_dist
                        min_dist := dist
                        closest_lvl := q.c
    closest_lvl

                    if array.size(q.labels) > 0
                        for j = 0 to array.size(q.labels) - 1
                            label.delete(array.get(q.labels, j))
                    array.remove(qml_zones, i)

f_in_golden_zone(p, top, bot, is_up) =>
    if na(top) or na(bot) or top == 0 or bot == 0
        false
    else
        float r = math.abs(top - bot)
        float max_p = math.max(top, bot)
        float min_p = math.min(top, bot)
        float lvl_618 = is_up ? (max_p - 0.618 * r) : (min_p + 0.618 * r)
        float lvl_886 = is_up ? (max_p - 0.886 * r) : (min_p + 0.886 * r)
        float max_l = math.max(lvl_618, lvl_886)
        float min_l = math.min(lvl_618, lvl_886)
        p <= max_l and p >= min_l

var array<line> fib_lines = array.new<line>()
var array<label> fib_labels = array.new<label>()
var array<linefill> fib_fills = array.new<linefill>()
var array<float> fib_ret_levels = array.from(0.0, 0.236, 0.382, 0.5, 0.618, 0.705, 0.786, 0.886, 1.0)
var array<float> fib_ext_levels = array.from(-0.236, -0.5, -0.618, -1.0, -1.5, -2.0, -2.5, -4.0)

f_draw_fib(_t1, _p1, _t2, _p2) =>
    float r = math.abs(_p1 - _p2)
    int st_t = int(math.min(_t1, _t2))
    bool is_up = (_t1 < _t2 and _p1 < _p2) or (_t1 > _t2 and _p1 > _p2)
    bool dir_up = fibReverse ? not is_up : is_up
    float high_p = math.max(_p1, _p2)
    float low_p = math.min(_p1, _p2)
    int cur_t = int(time)
    int en_t = int(math.max(_t1, _t2))
    int cur_b = bar_index

    // Yön rengi hesaplama
    color dir_clr = fibUseDirColor ? (dir_up ? fibBullColor : fibBearColor) : na

    if fibShowTrend
        color tr_clr = fibUseDirColor ? color.new(dir_clr, 40) : color.new(fibTrendColor, 40)
        line ln_tr = line.new(x1=int(_t1), y1=_p1, x2=int(_t2), y2=_p2, xloc=xloc.bar_time, color=tr_clr, style=line.style_dashed)
        line.set_width(ln_tr, 1)
        array.push(fib_lines, ln_tr)
    if fibShowCoord
        color coord_clr = fibUseDirColor ? dir_clr : (is_up ? style_fib_premium : style_fib_discount)
        array.push(fib_labels, label.new(x=int(_t2), y=_p2, text=str.tostring(_p2, "#.##"), xloc=xloc.bar_time, style=is_up?label.style_label_down:label.style_label_up, color=color.new(coord_clr, 80), textcolor=coord_clr, size=fib_lbl_sz))
        array.push(fib_labels, label.new(x=int(_t1), y=_p1, text=str.tostring(_p1, "#.##"), xloc=xloc.bar_time, style=is_up?label.style_label_up:label.style_label_down, color=color.new(coord_clr, 80), textcolor=coord_clr, size=fib_lbl_sz))

    // Etiket pozisyon hesaplama
    int lbl_x = fibLabelSide == "Sağ" ? cur_b + 5 : fibLabelSide == "Sol" ? math.max(0, cur_b - 200) : cur_b
    string lbl_ha = fibLabelSide == "Sağ" ? text.align_left : fibLabelSide == "Sol" ? text.align_right : text.align_center

    // Önceki seviye çizgisini saklama (background fill için)
    line prev_ln = na

    for i = 0 to array.size(fib_ret_levels) - 1
        float lvl = array.get(fib_ret_levels, i)
        float prc = dir_up ? (high_p - lvl * r) : (low_p + lvl * r)
        
        // Renk hesaplama
        color clr = na
        if fibUseDirColor
            clr := dir_clr
        else
            clr := style_fib_eq
            if lvl == 0.0 or lvl == 1.0
                clr := style_fib_boundary
            else if lvl == 0.5
                clr := style_fib_eq
            else if lvl >= 0.618
                clr := dir_up ? style_fib_discount : style_fib_premium
            else if lvl <= 0.382
                clr := dir_up ? style_fib_premium : style_fib_discount
        
        color f_clr = color.new(clr, style_fib_alpha)
        
        line ln_ret = line.new(x1=st_t, y1=prc, x2=en_t, y2=prc, xloc=xloc.bar_time, extend=fib_ext_mode, color=f_clr, style=fib_ls)
        int lw = (lvl == 0.0 or lvl == 0.5 or lvl == 0.618 or lvl == 0.786 or lvl == 1.0) ? 2 : 1
        line.set_width(ln_ret, lw)
        array.push(fib_lines, ln_ret)

        // Background fill
        if fibShowBg and not na(prev_ln)
            color bg_clr = color.new(nz(clr, color.gray), fibBgAlpha)
            linefill lf = linefill.new(prev_ln, ln_ret, bg_clr)
            array.push(fib_fills, lf)
        prev_ln := ln_ret
        
        // Etiket metni oluşturma
        string txt = ""
        if fibLevelDisplay == "Değer"
            txt := str.tostring(lvl, "#.###")
        else if fibLevelDisplay == "Yüzde"
            txt := str.tostring(lvl * 100, "#.#") + "%"
        else
            txt := str.tostring(lvl, "#.###") + " (" + str.tostring(lvl * 100, "#.#") + "%)"
        if fibShowPrice
            txt += "  " + str.tostring(prc, "#.##")
        array.push(fib_labels, label.new(x=lbl_x, y=prc, text=txt, xloc=xloc.bar_index, style=label.style_none, color=color.new(color.black, 100), textcolor=f_clr, size=fib_lbl_sz, textalign=lbl_ha))

    if fibShowExt
        for i = 0 to array.size(fib_ext_levels) - 1
            float lvl = array.get(fib_ext_levels, i)
            float prc = dir_up ? (high_p - lvl * r) : (low_p + lvl * r)
            color ext_clr = fibUseDirColor ? color.new(dir_clr, math.min(100, style_fib_alpha + 15)) : color.new(style_fib_ext, style_fib_alpha)
            
            line ln_ext = line.new(x1=st_t, y1=prc, x2=en_t, y2=prc, xloc=xloc.bar_time, extend=fib_ext_mode, color=ext_clr, style=line.style_dashed)
            line.set_width(ln_ext, 1)
            array.push(fib_lines, ln_ext)
            
            string txt = ""
            if fibLevelDisplay == "Değer"
                txt := str.tostring(lvl, "#.###")
            else if fibLevelDisplay == "Yüzde"
                txt := str.tostring(lvl * 100, "#.#") + "%"
            else
                txt := str.tostring(lvl, "#.###") + " (" + str.tostring(lvl * 100, "#.#") + "%)"
            if fibShowPrice
                txt += "  " + str.tostring(prc, "#.##")
            array.push(fib_labels, label.new(x=lbl_x, y=prc, text=txt, xloc=xloc.bar_index, style=label.style_none, color=color.new(color.black, 100), textcolor=ext_clr, size=fib_lbl_sz, textalign=lbl_ha))


if barstate.islast
    while array.size(fib_fills) > 0
        linefill.delete(array.pop(fib_fills))
    while array.size(fib_lines) > 0
        line.delete(array.pop(fib_lines))
    while array.size(fib_labels) > 0
        label.delete(array.pop(fib_labels))

    // MANUEL FIB ÇİZİMLERİ
    // MANUEL FIBONACCİ ÇİZİMLERİ
    if fibMan1Show and fm1_t1 != 0 and fm1_t2 != 0
        f_draw_fib(fm1_t1, fm1_p1, fm1_t2, fm1_p2)
    if fibMan2Show and fm2_t1 != 0 and fm2_t2 != 0
        f_draw_fib(fm2_t1, fm2_p1, fm2_t2, fm2_p2)
    if fibMan3Show and fm3_t1 != 0 and fm3_t2 != 0
        f_draw_fib(fm3_t1, fm3_p1, fm3_t2, fm3_p2)

    // OTOMATİK FIB ÇİZİMLERİ (Her TF İçin Sabit 2 Fib: Canlı Makro + Kırılan Makro)
    if fibAutoMainShow
        if not na(f3_l_t) and not na(f3_l_b)
            f_draw_fib(f3_l_bt, f3_l_b, f3_l_tt, f3_l_t)
        if fibShowBroken and not na(f3_b_t) and not na(f3_b_b)
            f_draw_fib(f3_b_bt, f3_b_b, f3_b_tt, f3_b_t)
    if fibAutoSubShow
        if not na(f2_l_t) and not na(f2_l_b)
            f_draw_fib(f2_l_bt, f2_l_b, f2_l_tt, f2_l_t)
        if fibShowBroken and not na(f2_b_t) and not na(f2_b_b)
            f_draw_fib(f2_b_bt, f2_b_b, f2_b_tt, f2_b_t)
    if fibAuto1mShow
        if not na(f1_l_t) and not na(f1_l_b)
            f_draw_fib(f1_l_bt, f1_l_b, f1_l_tt, f1_l_t)
        if fibShowBroken and not na(f1_b_t) and not na(f1_b_b)
            f_draw_fib(f1_b_bt, f1_b_b, f1_b_tt, f1_b_t)

    // QML Çizimleri yukarıda QMLZone state management ile halledildi.
// ═══════════════════════════════════════════════════════════
// BÖLÜM 17: DASHBOARD

var table_position = switch dashPos
    "Sol Üst" => position.top_left
    "Sağ Üst" => position.top_right
